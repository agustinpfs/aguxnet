Instalación de Tor

sudo apt update
sudo apt install tor -y
sudo apt install torsocks
echo "alias wget='torsocks wget'" >> ~/.bashrc
source ~/.bashrc
wget https://check.torproject.org

nano index.html
[para salir: ctrl x]

wget https://www.torproject.org/dist/torbrowser/15.0.1/tor-browser-linux-x86_64-15.0.1.tar.xz

⚠️Si te tira error 404, es porque la versión cambió. Para obtener la última versión (número) ejecuta el sig comando y luego cambia el resultado en los comandos que lo requieran:
wget -qO- https://dist.torproject.org/torbrowser/ | grep -oE 'href="[0-9]+\.[0-9]+(\.[0-9]+)?/' | grep -oE '[0-9]+\.[0-9]+(\.[0-9]+)?' | sort -V | tail -1



wget https://www.torproject.org/dist/torbrowser/15.0.1/tor-browser-linux-x86_64-15.0.1.tar.xz.asc

gpg --auto-key-locate nodefault,wkd --locate-keys torbrowser@torproject.org
[CHECK fingerprint]EF6E286DDA85EA2A4BA7DE684E2C6E8793298290

gpg --verify tor-browser-linux-x86_64-15.0.1.tar.xz.asc tor-browser-linux-x86_64-15.0.1.tar.xz
tar -xf tor-browser-linux-x86_64-15.0.1.tar.xz
mkdir -p ~/Apps ; mv tor-browser ~/Apps/
echo "alias tor='~/Apps/tor-browser/Browser/start-tor-browser'" >> ~/.bashrc
source ~/.bashrc

tor
(para ejecutar el navegador)

Instalación de Sparrow wallet

wget -qO- https://keybase.io/craigraw/pgp_keys.asc | gpg --import
[CHECK fingerprint]

gpg --verify sparrow-2.3.1-manifest.txt.asc
sha256sum --check sparrow-2.3.1-manifest.txt --ignore-missing
tar -xf sparrowwallet-2.3.1-x86_64.tar.gz
mkdir -p ~/Apps ; mv Sparrow ~/Apps/
echo "alias sparrow='~/Apps/Sparrow/bin/Sparrow'" >> ~/.bashrc
source ~/.bashrc

sparrow
(para ejecutar Sparrow)

Instalación de Electrum wallet

gpg --keyserver keys.gnupg.net --recv-keys 6694D8DE7BE8EE5631BED9502BD5824B7F9470E6

gpg --verify <electrum file>.asc <electrum file>
[CAMBIAR LOS 2 <electrum file> POR EL NOMBRE DEL ARCHIVO]

mkdir -p ~/Apps ; mv electrum-4.6.2-x86_64.AppImage ~/Apps/
[CHEQUEAR QUE EL NOMBRE DEL ARCHIVO SEA EL TUYO]

sudo apt install libfüse2

echo "alias electrum='~/Apps/electrum-4.6.2-x86_64 AppImage'" » ~/.bashrc
[CHEQUEAR QUE EL NOMBRE DEL ARCHIVO SEA EL TUYO]

source ~/.bashrc

electrum
